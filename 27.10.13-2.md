#Zadanie 3
Tablica poteg liczby 2 oraz jej wyswietlanie

```c
#include <stdio.h>
#include <math.h>

int main ()
{
int a,b, tablica[5] = {0};
int i=0;
a=2;
tablica[0]=pow(a,0);
tablica[1]=pow(a,1);
tablica[2]=pow(a,2);
tablica[3]=pow(a,2)*2;
tablica[4]=pow(a,2)*pow(a,2);
for (i=0; i!=5; ++i)
printf("2 do potegi[%d]=%d\n", i, tablica[i]);
return 0;
}
```
